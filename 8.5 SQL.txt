SELECT I.ID, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO, L.NAME
FROM ISSUES I
JOIN ISSUESLIST L ON I.ISSUESLIST_ID = L.ID;

SELECT I.ID, I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U. LASTNAME
FROM ISSUES I
JOIN USERS U ON I.ISSUESLIST_ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
